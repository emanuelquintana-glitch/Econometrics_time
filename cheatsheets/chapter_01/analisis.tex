\section{Simulación de Monte Carlo}
En este bloque, R genera una muestra aleatoria y LaTeX la reporta en tiempo real.
No estamos escribiendo datos; estamos programando conocimiento.

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlcom{# Generación de datos interplanetarios}
\hlkwd{set.seed}\hldef{(}\hlnum{2026}\hldef{)}
\hldef{n} \hlkwb{<-} \hlnum{100}
\hldef{presupuesto_nasa} \hlkwb{<-} \hlkwd{rnorm}\hldef{(n,} \hlkwc{mean} \hldef{=} \hlnum{500}\hldef{,} \hlkwc{sd} \hldef{=} \hlnum{50}\hldef{)}
\hldef{misiones_exito} \hlkwb{<-} \hlnum{0.8} \hlopt{*} \hldef{presupuesto_nasa} \hlopt{+} \hlkwd{rnorm}\hldef{(n,} \hlkwc{sd} \hldef{=} \hlnum{20}\hldef{)}

\hlcom{# Regresión lineal automática}
\hldef{modelo} \hlkwb{<-} \hlkwd{lm}\hldef{(misiones_exito} \hlopt{~} \hldef{presupuesto_nasa)}
\hlkwd{coef}\hldef{(modelo)}
\end{alltt}
